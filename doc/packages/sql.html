2.1. Raw SQL queries

This section contains examples of all important methods

2.1.1. Query

Query

$sql = new PSX_Sql('localhost', 'root', '', 'psx');

// query
$query = 'INSERT INTO `news` SET `title` = ?, `date` = NOW()';

$sql->query($query, array('Foobar'));

2.1.2. Select all

Select all

$sql = new PSX_Sql('localhost', 'root', '', 'psx');

$query = 'SELECT `id`, `title`, `date` FROM `news` WHERE `id` > ? LIMIT 8';

// assoc array
$result = $sql->getAll($query, array(6));

foreach($result as $row)
{
	echo $row['title'] . "\n";
}

// object array
$result = $sql->getAll($query, array(6), PSX_Sql::FETCH_OBJECT);

foreach($result as $row)
{
	echo $row->title . "\n";
}

// object array with custom class
$result = $sql->getAll($query, array(6), PSX_Sql::FETCH_OBJECT, 'My_Record', array('Constructor', 'Args'));

foreach($result as $row)
{
	echo $row->title . "\n";
}

2.1.3. Select row

Select row

$sql = new PSX_Sql('localhost', 'root', '', 'psx');

$query = 'SELECT `id`, `title`, `date` FROM `news` WHERE `id` = ?';

// array
$row = $sql->getRow($query, array(6));

echo $row['title'] . "\n";

// object
$row = $sql->getRow($query, array(6), PSX_Sql::FETCH_OBJECT);

echo $row->title . "\n";

// custom class
$row = $sql->getRow($query, array(6), PSX_Sql::FETCH_OBJECT, 'My_Record', array('Constructor', 'Args'));

echo $row->title . "\n";
